int pinoSensor = 1;
int valorLido = 0;
float temperatura = 0;

int CODIGO = 00000; // coloca aqui  o código do arduino

void setup() {
  Serial.begin(9600);}

void loop() {
valorLido = analogRead(pinoSensor);
temperatura = (valorLido * 0.00488);
temperatura = temperatura * 100;
Serial.print(temperatura);
Serial.print("-");
Serial.print(CODIGO);
Serial.println();
delay(1000);
}
